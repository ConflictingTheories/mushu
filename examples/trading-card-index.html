Quick Start

// Import the trading card system import { tradingCardGenerator, createCardViewer } from './trading-card.js'; // Configure your card const card = tradingCardGenerator() .setCardType('bordered') .setLayoutStyle('portrait') .setFrontImage('image.jpg') .setBorderColor('#4a00e0') .setHologramIntensity(0.7); // Create viewer const viewer = await createCardViewer('#canvas', card.getConfig()); // Control the card viewer.flipCard(true); viewer.setHologramIntensity(0.8);
